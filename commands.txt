# ── Cloudflare Tunnel ──────────────────────────────────────────────────

# Quick Tunnel (sin dominio, URL aleatoria, ideal para pruebas)
# 1. Levantar infraestructura
docker compose up -d
# 2. Iniciar backend
poetry run uvicorn app.app:app --reload --port 8000
# 3. En otra terminal, abrir túnel efímero
cloudflared tunnel --url http://localhost:8000
# → Usar la URL que aparezca (ej: https://rana-llama-123.trycloudflare.com)

# Managed Tunnel (con dominio propio, para entrega)
cloudflared tunnel login
cloudflared tunnel create taller5-tunnel
cloudflared tunnel route dns taller5-tunnel sensor-api.midominio.com
echo "CF_TUNNEL_TOKEN=<token_del_json>" >> .env
echo "CF_TUNNEL_DOMAIN=https://sensor-api.midominio.com" >> .env
docker compose up -d
poetry run uvicorn app.app:app --reload --port 8000
